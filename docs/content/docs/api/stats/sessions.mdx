## Get Sessions

<div className="grid grid-cols-1 xl:grid-cols-2 gap-8">
<div>

```
GET /api/sessions/:site
```

Returns a paginated list of sessions with their analytics data.

### Path Parameters

<TypeTable
  type={{
    site: {
      description: 'Site ID or site identifier',
      type: 'string | number',
      required: true,
    },
  }}
/>

### Query Parameters

Accepts all [Common Parameters](/docs/api/getting-started#common-parameters) plus the following:

<TypeTable
  type={{
    page: {
      description: 'Page number (1-indexed)',
      type: 'number',
      default: 1,
    },
    limit: {
      description: 'Number of results per page',
      type: 'number',
      default: 100,
    },
    user_id: {
      description: 'Filter sessions by user ID',
      type: 'string',
      required: false,
    },
    identified_only: {
      description: 'Only return sessions from identified users',
      type: 'string',
      typeDescription: '"true" | "false"',
      default: 'false',
    },
  }}
/>

### Response

<TypeTable
  type={{
    data: {
      description: 'Array of session objects',
      type: 'Session[]',
    },
  }}
/>

<TypeTable

    session_id: {
      description: 'Unique session identifier',

      description: 'Custom user ID when identified',
      type: 'string',

    traits: {
      description: 'Custom user traits',

      description: 'Country code',

      description: 'Region/state',

    operating_system: {
      description: 'Operating system',

## Get Session

<div className="grid grid-cols-1 xl:grid-cols-2 gap-8">
<div>

```
GET /api/sessions/:sessionId/:site
```

Returns detailed information about a specific session including all events and pageviews.

### Path Parameters

<TypeTable
  type={{
    sessionId: {
      description: 'Session ID',
      type: 'string',
      required: true,
    },
    site: {
      description: 'Site ID or site identifier',
      type: 'string | number',
      required: true,
    },
  }}
/>

### Query Parameters

<TypeTable
  type={{
    limit: {
      description: 'Number of events to return',
      type: 'number',
      default: 100,
    },
    offset: {
      description: 'Number of events to skip',
      type: 'number',
      default: 0,
    },
    minutes: {
      description: 'Only return events from the last N minutes',
      type: 'number',
      required: false,
    },
  }}
/>

### Response

<TypeTable
  type={{
    data: {
      description: 'Session details object',
      type: 'SessionDetails',
    },
  }}
/>

#### SessionDetails Object

<TypeTable
  type={{
    session: {
      description: 'Session metadata',
      type: 'SessionMeta',
    },
    events: {
      description: 'Array of events in the session',
      type: 'Event[]',
    },
    pagination: {
      description: 'Pagination info for events',
      type: 'Pagination',
    },
  }}
/>

#### Event Object

<TypeTable
  type={{
    timestamp: {
      description: 'Event timestamp',
      type: 'string',
      typeDescription: 'ISO 8601 datetime',
    },
    pathname: {
      description: 'Page path',
      type: 'string',
    },
    hostname: {
      description: 'Hostname',
      type: 'string',
    },
    page_title: {
      description: 'Page title',
      type: 'string',
    },
    type: {
      description: 'Event type',
      type: 'string',
      typeDescription: '"pageview" | "custom_event" | "error" | "outbound"',
    },
    event_name: {
      description: 'Custom event name (for custom_event type)',
      type: 'string',
    },
    props: {
      description: 'Event properties',
      type: 'object',
    },
  }}
/>

</div>
