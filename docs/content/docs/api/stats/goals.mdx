## Get Goals

<div className="grid grid-cols-1 xl:grid-cols-2 gap-8">
<div>

```
GET /api/goals/:site
```

Returns a paginated list of goals with their conversion metrics.

### Path Parameters

<TypeTable
  type={{
    site: {
      description: 'Site ID or site identifier',
      type: 'string | number',
      required: true,
    },
  }}
/>

### Query Parameters

Accepts all [Common Parameters](/docs/api/getting-started#common-parameters) plus the following:

<TypeTable
  type={{
    page: {
      description: 'Page number (1-indexed)',
      type: 'number',
      default: 1,
    },
    page_size: {
      description: 'Number of results per page (max 100)',
      type: 'number',
      default: 10,
    },
    sort: {
      description: 'Field to sort by',
      type: 'string',
      typeDescription: '"goalId" | "name" | "goalType" | "createdAt"',
      default: 'createdAt',
    },
    order: {
      description: 'Sort direction',
      type: 'string',
      typeDescription: '"asc" | "desc"',
      default: 'desc',
    },
  }}
/>

### Response

<TypeTable
  type={{
    data: {
      description: 'Array of goal objects with conversion data',
      type: 'Goal[]',
    },
    meta: {
      description: 'Pagination metadata',
      type: 'PaginationMeta',
    },
  }}
/>

#### Goal Object

<TypeTable
  type={{
    goalId: {
      description: 'Unique goal identifier',
      type: 'number',
    },
    name: {
      description: 'Goal name',
      type: 'string | null',
    },
    goalType: {
      description: 'Type of goal',
      type: 'string',
      typeDescription: '"path" | "event"',
    },
    config: {
      description: 'Goal configuration',
      type: 'GoalConfig',
    },
    createdAt: {
      description: 'Creation timestamp',
      type: 'string',
      typeDescription: 'ISO 8601 datetime',
    },
    total_conversions: {
      description: 'Number of sessions that completed the goal',
      type: 'number',
    },
    total_sessions: {
      description: 'Total sessions in the time period',
      type: 'number',
    },
    conversion_rate: {
      description: 'Conversion rate (0-1)',
      type: 'number',
    },
  }}
/>

#### GoalConfig Object

<TypeTable
  type={{
    pathPattern: {

      description: 'Event name to match (for event goals)',
      type: 'string',

      description: 'Event property key to match',

## Create Goal

<div className="grid grid-cols-1 xl:grid-cols-2 gap-8">
<div>

```
POST /api/goals/:site
```

Creates a new goal for tracking conversions.

### Path Parameters

<TypeTable
  type={{
    site: {
      description: 'Site ID or site identifier',
      type: 'string | number',
      required: true,
    },
  }}
/>

### Request Body

<TypeTable
  type={{
    name: {
      description: 'Goal name',
      type: 'string',
      required: false,
    },
    goalType: {
      description: 'Type of goal',
      type: 'string',
      typeDescription: '"path" | "event"',
      required: true,
    },
    config: {
      description: 'Goal configuration',
      type: 'GoalConfig',
      required: true,
    },
  }}
/>

#### GoalConfig for Path Goals

<TypeTable
  type={{
    pathPattern: {
      description: 'URL path pattern to match. Supports wildcards (*)',
      type: 'string',
      required: true,
      typeDescription: 'e.g., "/checkout/*", "/signup/complete"',
    },
  }}
/>

#### GoalConfig for Event Goals

<TypeTable
  type={{
    eventName: {
      description: 'Event name to match',
      type: 'string',
      required: true,
    },
    eventPropertyKey: {
      description: 'Property key to match (optional)',
      type: 'string',
      required: false,
    },
    eventPropertyValue: {
      description: 'Property value to match (required if key is provided)',
      type: 'string | number | boolean',
      required: false,
    },
  }}
/>

### Response

<TypeTable
  type={{
    success: {
      description: 'Whether the goal was created',
      type: 'boolean',
    },
    goalId: {
      description: 'ID of the created goal',
      type: 'number',
    },
  }}
/>
